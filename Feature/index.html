Feature

An element for featuring other content

Children: Hero Highlight

Properties

image Image The main image
heading Heading The largest heading
subheading Subheading The smaller heading
body Body text The smaller text under the headings
link Link The URL redirected to when clicked

Definition

JSON 
{
    "@context": "http://uischema.org",
    "@type": "Feature",
    "@label": "heading",
    "@topic": [ "media", "text" ],
    "image": "Image",
    "heading": "string",
    "subheading": "string",
    "body": "string",
    "link": "Link"
}
Template 
<div data-ui-schema="uischema.org/Feature" class="ui-feature">
    {{# image }}
        <div class="ui-feature__image">
            {{> Image }}
        </div>
    {{/ image }}

    <div class="ui-feature__details">
        {{# heading }}
            <h3 class="ui-feature__heading">{{ . }}</h3>
        {{/ heading }}
        
        {{# subheading }}
            <h4 class="ui-feature__subheading">{{ . }}</h4>
        {{/ subheading }}
        
        {{# body }}
            <div class="ui-feature__body">{{{ . }}}</div>
        {{/ body }}

        {{# link }}
            <div class="ui-feature__link">{{> Link }}</div>
        {{/ link }}
    </div>
</div>
